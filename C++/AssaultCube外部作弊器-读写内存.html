DWORD getProcId(const wchar_t* procName) {
	DWORD procID = 0;
	// 进程快照句柄
	HANDLE hProcessSnap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
	if (hProcessSnap == INVALID_HANDLE_VALUE) {
		printf("CreateToolhelp32Snapshot error.\n");
		return 0;
	}
	//存放进程快照
	PROCESSENTRY32 procEntry;
	procEntry.dwSize = sizeof(PROCESSENTRY32);
	if (Process32First(hProcessSnap, &procEntry)) {
		do {
			printf("%ls----%d\n", procEntry.szExeFile, procEntry.th32ProcessID);
			//比较两个Unicode字符串 = 0 为相等
			if (!_wcsicmp(procEntry.szExeFile, procName)) {
				procID = procEntry.th32ProcessID;
				break;
			}
		} while (Process32Next(hProcessSnap, &procEntry));
	}

	CloseHandle(hProcessSnap);
	return procID;
}
uintptr_t getModuleBaseAddress(DWORD procId, const wchar_t* modName)
{
	uintptr_t modBaseAddr = 0;
	HANDLE hSnap = CreateToolhelp32Snapshot(TH32CS_SNAPMODULE | TH32CS_SNAPMODULE32, procId);
	if (hSnap != INVALID_HANDLE_VALUE)
	{
		MODULEENTRY32 modEntry;
		modEntry.dwSize = sizeof(modEntry);
		if (Module32First(hSnap, &modEntry))
		{
			do
			{
				if (!_wcsicmp(modEntry.szModule, modName))
				{
					modBaseAddr = (uintptr_t)modEntry.modBaseAddr;
					break;
				}
			} while (Module32Next(hSnap, &modEntry));
		}
	}
	CloseHandle(hSnap);
	return modBaseAddr;
}
uintptr_t findDMAAddy(HANDLE hProc, uintptr_t ptr, std::vector<unsigned int> offsets)
{
	uintptr_t addr = ptr;
	for (unsigned int i = 0; i < offsets.size(); ++i)
	{
		ReadProcessMemory(hProc, (BYTE*)addr, &addr, sizeof(addr), 0);
		addr += offsets[i];
	}
	return addr;
}

内存相关操作

void mem::PatchEx(BYTE* dst, BYTE* src, unsigned int size, HANDLE hProcess) {
	//接收指定页区域中第一页的上一个访问权限
	DWORD oldProtect;
	//改变这段内存的权限为读写
	VirtualProtectEx(hProcess,dst,size,PAGE_EXECUTE_READWRITE,&oldProtect);
	//写内存替换指令
	WriteProcessMemory(hProcess, dst, src, size, nullptr);
	//写回权限
	VirtualProtectEx(hProcess, dst, size, oldProtect, &oldProtect);

}
void mem::NopEx(BYTE* dst, unsigned int size, HANDLE hProcess) {
	BYTE* nopArray = new BYTE[size];
	//写入nop指令，对应字节为90
	memset(nopArray, 0x90, size);
	//替换指令
	PatchEx(dst, nopArray, size, hProcess);
	//释放内存
	delete[] nopArray;
}

功能实现

//获取进程id
DWORD procId = getProcId(L"ac_client.exe");

//取进程句柄
HANDLE hProcess = 0;
hProcess = OpenProcess(PROCESS_ALL_ACCESS, NULL, procId);
uintptr_t moduleBase = 0, localPlayerPtr = 0, healthAddr = 0;
//锁定血量、子弹、后坐力的标识
bool bHealth = false, bAmmo = false, bRecoil = false;
const int newValue = 995;

if (!procId) {
    std::cout << "Process not found, press enter to exit\n";
    return 0;
}
//取进程句柄
hProcess = OpenProcess(PROCESS_ALL_ACCESS, NULL, procId);

//取模块地址
moduleBase = getModuleBaseAddress(procId, L"ac_client.exe");

localPlayerPtr = moduleBase + 0x10f4f4;

healthAddr = findDMAAddy(hProcess, localPlayerPtr, { 0xF8 });


DWORD dwExit = 0;
while (GetExitCodeProcess(hProcess, &dwExit) && dwExit == STILL_ACTIVE) {
    //持续写入血量，J
    if (GetAsyncKeyState(0x4A) & 1)
    {
        bHealth = !bHealth;
    }

    //无限子弹补丁，k
    if (GetAsyncKeyState(0x4B) & 1)
    {
        bAmmo = !bAmmo;

        if (bAmmo)
        {
            //FF 06 = inc [esi]
            mem::PatchEx((BYTE*)(moduleBase + 0x637e9), (BYTE*)"\xFF\x06", 2, hProcess);
        }

        else
        {
            //FF 0E = dec [esi]
            mem::PatchEx((BYTE*)(moduleBase + 0x637e9), (BYTE*)"\xFF\x0E", 2, hProcess);
        }
    }

    //无后座nop，L
    if (GetAsyncKeyState(0x4C) & 1)
    {
        bRecoil = !bRecoil;

        if (bRecoil)
        {
            mem::NopEx((BYTE*)(moduleBase + 0x63786), 10, hProcess);
        }

        else
        {
            //50 8D 4C 24 1C 51 8B CE FF D2; 
            mem::PatchEx((BYTE*)(moduleBase + 0x63786), (BYTE*)"\x50\x8D\x4C\x24\x1C\x51\x8B\xCE\xFF\xD2", 10, hProcess);
        }
    }
    //退出
    if (GetAsyncKeyState(VK_END) & 1)
    {
        return 0;
    }

    //写入血量
    if (bHealth)
    {
        mem::PatchEx((BYTE*)healthAddr, (BYTE*)&newValue, sizeof(newValue), hProcess);
    }

    Sleep(10);
}
